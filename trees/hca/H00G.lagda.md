{-# OPTIONS --cubical #-}

open import Agda.Primitive using (Level; lzero; Set)
open import Agda.Builtin.Nat using (Nat; zero; suc)
open import Agda.Builtin.List using (List; []; _∷_)
open import Agda.Builtin.Unit using (⊤; tt)

-- Stage-indexed forms
data Form : Nat → Set lzero where
  mk : {n : Nat} → (L R : List (Form n)) → Form (suc n)

length : {A : Set lzero} → List A → Nat
length []       = zero
length (_ ∷ xs) = suc (length xs)

-- inductive ordering on Naturals
data _<n_ : Nat → Nat → Set lzero where
  z<s : {m : Nat} → zero <n suc m
  s<s : {n m : Nat} → n <n m → suc n <n suc m

-- Toy ordering on Form
data _<_ {n : Nat} : Form (suc n) → Form (suc n) → Set lzero where
  byLeftLength :
    {L R L' R' : List (Form n)} →
    length L <n length L' →
    mk {n} L R < mk {n} L' R'
